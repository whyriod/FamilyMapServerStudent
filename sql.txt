Create Table Authtoken(
    AuthToken            varchar(20) UNIQUE NOT NULL,
    AssociatedUsername   varchar(20) NOT NULL
);

Create Table Event(
    EventID              varchar(20) PRIMARY KEY NOT NULL,
    EventType            varchar(20) check(EventType in ('birth', 'baptism', 'christening', 'marriage', 'death')),
    PersonID             varchar(20) NOT NULL Check (PersonID != ""),
    AssociatedUsername   varchar(20) NOT NULL,
    Year                 Integer NOT NULL,
    Country              varchar(50) NOT NULL,
    City                 varchar(50) NOT NULL,
    Latitude             float NOT NULL,
    Longitude            float NOT NULL
);

Create Table Person(
    PersonID             varChar(20) PRIMARY KEY check (PersonID != ""),
    AssociatedUsername   varchar(20) NOT NULL UNIQUE DEFAULT "",
    FirstName            varchar(40) NOT NULL,
    LastName             varChar(40) NOT NULL,
    Gender               varchar(1) NOT NULL Check (Gender in ('M','F')),
    FatherID             varChar(20),
    MotherID             varChar(20),
    SpouseID             varChar(20),
);

Create Table User (
    PersonID             varChar(20) UNIQUE NOT NULL check (PersonID != ""),
    Username             varchar(20) PRIMARY KEY,
    Password             varchar(20) NOT NULL,
    Email                varchar(40) NOT NULL,
    FirstName            varchar(40) NOT NULL,
    LastName             varchar(40) NOT NULL,
    Gender               varchar(1) NOT NULL Check (Gender in ('M','F'))
);
