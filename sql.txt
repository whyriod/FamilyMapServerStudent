Create Table User (
    PersonID             varChar(20) PRIMARY KEY AUTOINCREMENT check (PersonID != ""),
    Username             varchar(20) UNIQUE NOT NULL,
    Password             varchar(20) NOT NULL,
    Email                varchar(40) NOT NULL,
    FirstName            varchar(40) NOT NULL,
    LastName             varchar(40) NOT NULL,
    Gender               varchar(1) NOT NULL Check (Gender in ('M','F'))
);

Create Table Person(
    PersonID             varChar(20) PRIMARY KEY AUTOINCREMENT check (PersonID != ""),
    AssociatedUsername   varchar(20) NOT NULL DEFAULT "",
    FirstName            varchar(40) NOT NULL,
    LastName             varChar(40) NOT NULL,
    Gender               varchar(1) NOT NULL Check (Gender in ('M','F')),
    FatherID             varChar(20) NOT NULL DEFAULT "", --Default to "" rather than null
    MotherID             varChar(20) NULL DEFAULT "", --Default to "" rather than null
    SpouseID             varChar(20) NULL DEFAULT ""  --Default to "" rather than null
);

Create Table Event(
    EventID              String PRIMARY KEY NOT NULL,
    EventType            String check(EventType in ('birth', 'baptism', 'christening', 'marriage', 'death')),
    PersonID             varchar(20) NOT NULL Check (PersonID != ""),
    AssociatedUsername   varchar(20) NOT NULL,
    Year                 Integer NOT NULL,
    Country              String NOT NULL,
    City                 String NOT NULL,
    Latitude             float NOT NULL,
    Longitude            float NOT NULL
);

Create Table Token(
    AuthToken           String UNIQUE NOT NULL
    AssociatedUsername   varchar(20) NOT NULL,
);
